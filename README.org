#  N個の数値の読み取り
n = int(input())
A = list(map(int, input().split()))

# 縦に長い数値の読み込み
a = []
for i in range(n):
    a.append(int(input()))

#+begin_src python
# N個の数値の書き出し
print(" ".join(map(str, A)))

# 縦に長い数値の書き出し
print("\n".join(map(str, A)))
